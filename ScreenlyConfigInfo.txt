Configure Screenly to work on our network


//  1.  Expand root file system
sudo raspi-config

// 2.  Change default password
sudo raspi-config

// 3.  Turn on SSH
sudo raspi-config

// 4. Change timezone
Sudo raspi-condfig


// ——————— REBOOT ————————


// 5.  Disable screenly-net-manager
   cd /etc/systemd/system
   sudo systemctl disable screenly-net-manager.service

// 6.  Manually setup wifi on pi
sudo nano /etc/network/interfaces

auto lo

iface lo inet loopback
iface eth0 inet dhcp

allow-hotplug wlan0
auto wlan0

iface wlan0 inet dhcp
        wpa-ssid “msdcart”
        wpa-psk “spongebob;”

// save and exit.
